package Step3;

public class MultiplyTable {
	public static void main(String args[]){
		/********** Begin *********/
		
		for(int i=1;i<=9;i++)
		{
            System.out.print(" ");
			for(int j=1;j<=9&&j<=i;j++)
			{
                if(i!=j)
					System.out.print(i+"×"+j+"="+i*j+" ");
                else
                    System.out.print(i+"×"+j+"="+i*j);
			}
			System.out.println();
		}
		 /********** End *********/
	}
}
