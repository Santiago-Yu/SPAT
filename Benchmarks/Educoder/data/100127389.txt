package step2;

public class Point {
	private double x;
	private double y;

	public Point() {
		this(0, 0);
	}

	public Point(double x, double y) {
		this.x = x;
		this.y = y;
		System.out.println("Point constructor:Center=" + this.toString()+"radius=0.0");
	}

	public String toString() {
		return "[" + this.x + "," + this.y + "]";
	}
    
    public double getX() {
		return x;
	}

	public void setX(double x) {
		this.x = x;
	}

	public double getY() {
		return y;
	}

	public void setY(double y) {
		this.y = y;
	}

}
